## Controlled
You can control the radio and the checkbox with the ```checked``` and ```onChange``` props.
<Canvas>
  <Story name="Controlled">
    {() => {
      const [checkedRadio, setCheckedRadio] = React.useState();
      const [checkedCheck, setCheckedCheck] = React.useState(true);
      return (
        <>
          <Check
            type="radio"
            value="custom-radio-1"
            checked={checkedRadio === "custom-radio-1"}
            onChange={(event) => setCheckedRadio(event.target.value)}
            id="custom-radio-1"/>
          <Check
            type="radio"
            value="custom-radio-2"
            checked={checkedRadio === "custom-radio-2"}
            onChange={(event) => setCheckedRadio(event.target.value)}
            id="custom-radio-2"/>
          <Check
            type="checkbox"
            checked={checkedCheck}
            onChange={() => setCheckedCheck(!checkedCheck)}
            id="custom-checkbox"/>
        </>
      );
    }}
  </Story>
</Canvas>
